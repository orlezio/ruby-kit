def as_html(link_resolver)
  block_group = ->(block){
    case block
    when Block::ListItem
      block.ordered? ? "ol" : "ul"
    else
      nil
    end
  }
  groups, last = [], nil
  blocks.each {|block|
    group = block_group.(block)
    groups << Group.new(group) if !last || group != last
    groups.last << block
    last = group
  }
  groups.map{|group|
    html = group.blocks.map{|b| b.as_html(link_resolver) }.join
    case group.kind
    when "ol"
      %(<ol>#{html}</ol>)
    when "ul"
      %(<ul>#{html}</ul>)
    else
      html
    end
  }.join("\n\n")
end
